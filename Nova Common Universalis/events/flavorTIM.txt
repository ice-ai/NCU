		tag = TIM
		exists = QOM
		has_ruler = "Shah Rukh"
	}

	mean_time_to_happen = {
		months = 1
	}
	
	option = { 
		name = "flavor_tim.0.a" #
		hidden_effect = {
			every_country = {
				limit = {
					NOT = { tag = ROOT }
					OR = {
						is_neighbor_of = ROOT
						dynasty = Timurid
						any_owned_province = {
							is_core = TIM
						}
						any_neighbor_country = {
							dynasty = Timurid
						}
					}
				}
				country_event = { id = flavor_tim.4 }
			}
		}
	}
}

# Death of Shah Rukh
country_event = {
	id = flavor_tim.1
	title = "flavor_tim.1.t"
	desc = "flavor_tim.1.d"
	picture = DEATH_OF_HEIR_eventPicture
	
	is_triggered_only = yes
	fire_only_once = yes
	
	trigger = {
		tag = TIM
		NOT = { has_global_flag = death_of_shah_rukh }
	}

	mean_time_to_happen = {
		days = 30
	}
	
	immediate = { 
		set_global_flag = death_of_shah_rukh 
	}
	
	option = { 
		name = "flavor_tim.1.a" # My time has come!
		if = {
			limit = {
				has_ruler = "Shah Rukh"
				ruler_has_personality = scholar_personality
				ruler_has_personality = architectural_visionary_personality
				ruler_has_personality = benevolent_personality
			}
			kill_ruler = yes
		}
		add_horde_unity = -30
		add_stability = -1
		custom_tooltip = flavor_tim.1.a.tt
		hidden_effect = {
			every_country = {
				limit = {
					NOT = { tag = ROOT }
					OR = {
						is_neighbor_of = ROOT
						dynasty = Timurid
						any_owned_province = {
							is_core = TIM
						}
						any_neighbor_country = {
							dynasty = Timurid
						}
					}
				}
				country_event = { id = flavor_tim.2 }
				if = {
					limit = { dynasty = Timurid }
					set_country_flag = Timurid_Heir
				}
			}
			set_country_flag = Timurid_Heir
			every_province = {
				limit = {
					owner = {
						OR = {
							tag = TIM
							AND = {
								is_subject_of = TIM
								NOT = { is_subject_of_type = tributary_state }
							}
						}
					}
				}
				set_province_flag = tim_traget_province
			}
		}
		every_country = {
			limit = {
				is_subject_of = TIM
				NOT = { is_subject_of_type = tributary_state }
			}
			grant_independence = yes	
		}
	}
}

# Death of Shah Rukh
country_event = {
	id = flavor_tim.2
	title = "flavor_tim.2.t"
	desc = "flavor_tim.2.d"
	picture = DEATH_OF_HEIR_eventPicture
	
	is_triggered_only = yes

	option = {
		name = "flavor_tim.2.a" # My time has come!
		trigger = { dynasty = Timurid }
		custom_tooltip = flavor_tim.2.a.tt
		custom_tooltip = flavor_tim.2.t.tt
		add_country_modifier = {
			name = fight_for_throne
			duration = 3650
		}
		if = {
			limit = { tag = TRS }
			add_country_modifier = {
				name = high_legitimacy
				duration = 3650
			}
		}
		set_country_flag = tim_traget_country
		hidden_effect = {
			if = {
				limit = { has_country_flag = Timurid_Heir }
				change_variable = { which = cu_tim_counter which = TIM }
			}
			every_province = {
				limit = {
					has_province_flag = tim_traget_province
				}
				add_permanent_claim = ROOT
			}
			capital_scope = {
				build_to_forcelimit = {
					infantry = 0.6
					cavalry = 0.3
				}
			}
			add_manpower = 10
		}
		every_country = {
			limit = {
				has_country_flag = Timurid_Heir
			}
			ROOT = {
				declare_war_with_cb = {
					who = PREV
					casus_belli = cb_tim_war
				}
			}
		}
	}
	option = {
		name = "flavor_tim.2.b" #
		trigger = { NOT = { dynasty = Timurid } }
		custom_tooltip = flavor_tim.2.a.tt
		if = {
			limit = {
				NOT = { dynasty = "Timurid" }
				government = monarchy
				NOT = { tag = HLR }
				NOT = { tag = ROM }
				NOT = { tag = MUG }
				NOT = { tag = PER }
				OR = {
					cu_iranian_group = yes
					primary_culture = uzbehk
					primary_culture = turkmeni
				}
				NOT = { primary_culture = azerbaijani }
				NOT = { has_reform =  tribal_federation }
				religion_group = muslim
			}
		custom_tooltip = flavor_tim.2.b.tt
		}
	}
}

	}
}
